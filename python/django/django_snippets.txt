# Delete Model 

obj = ModelName.objects.all()

while obj.count() > 9:
	obj.first().delete()

while obj.count() > 9:
	obj.last().delete()
	
while obj.count() > 9:
	mid = obj.count() // 2
	obj[mid].delete()
	
# django get all related of models	
[
	f for f in ModelName._meta.get_fields()
	if (f.one_to_many or f.one_to_one)
	   and f.auto_created and not f.concrete
]
